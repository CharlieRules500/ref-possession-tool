    <div>

      <input id="team1Name" type="text" placeholder="Team 1 Name" />

      <input id="team1Color" type="color" value="#1e90ff" />

    </div>

    <div>

      <input id="team2Name" type="text" placeholder="Team 2 Name" />

      <input id="team2Color" type="color" value="#ff5733" />

    </div>

    <button id="apply">Apply Names & Colors</button>

    <button id="reset">Reset Game</button>

    <div id="quarter">Quarter: 1</div>

  </div>


  <script>

    const team1Div = document.getElementById("team1");

    const team2Div = document.getElementById("team2");

    const quarterDisplay = document.getElementById("quarter");


    let quarter = 1;

    let teamInPossession = null;


    function setPossession(team) {

      if (team === "team1") {

        teamInPossession = "team1";

        team1Div.classList.remove("inactive");

        team2Div.classList.add("inactive");

      } else {

        teamInPossession = "team2";

        team2Div.classList.remove("inactive");

        team1Div.classList.add("inactive");

      }

      quarterDisplay.textContent = "Quarter: " + quarter;

      quarter++;

    }


    team1Div.addEventListener("click", () => setPossession("team1"));

    team2Div.addEventListener("click", () => setPossession("team2"));


    document.getElementById("apply").addEventListener("click", () => {

      const name1 = document.getElementById("team1Name").value || "Team 1";

      const name2 = document.getElementById("team2Name").value || "Team 2";

      const color1 = document.getElementById("team1Color").value || "#1e90ff";

      const color2 = document.getElementById("team2Color").value || "#ff5733";


      team1Div.textContent = name1;

      team2Div.textContent = name2;

      team1Div.style.backgroundColor = color1;

      team2Div.style.backgroundColor = color2;

    });


    document.getElementById("reset").addEventListener("click", () => {

      team1Div.classList.remove("inactive");

      team2Div.classList.remove("inactive");

      quarter = 1;

      teamInPossession = null;

      quarterDisplay.textContent = "Quarter: 1";

    });

  </script>
